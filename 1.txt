class Solution {
public:
    string removeOccurrences(string s, string part) {
        int i = 0;
        int j = 0;
        int ind = -1;
        int n = part.length();
        while(i<s.length() && j<n){
            if(s[i]==part[j]){
                if(ind==-1) ind = i;
                if(j==n-1){
                    s.erase(ind,n);
                    i=0;
                    j=0;
                    ind = -1;
                }else{
                    i++;
                    j++;
                }
            }else{
                if(j>0){
                    i=ind+1;
                    j=0;
                    ind = -1;
                }else i++;
            }
        }

        return s;
    }

// minimum in a sorted sorted array

int findSmallest(vector<int>& gems) { 
    int n = gems.size();
    int low = 0;
    int high = n-1;
    while(high>low){
        int mid = low + (high-low)/2;
        if(gems[mid]>gems[high]){ // minimum in right side
            low = mid+1;
        }else{
            high = mid;
        }
    }
    return gems[low];
}

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        int n = nums.size();
        sort(nums.begin(),nums.end());
        set<vector<int>> ans;
        for(int i=0;i<n;i++){
            int j=i+1; // left
            int k=n-1; // right
            while(k>j){
                int sum = nums[i]+nums[j]+nums[k];
                if(sum==0){
                    ans.insert({nums[i],nums[j],nums[k]});
                    j++;
                    k--;
                } 
                else if(sum>0) k--;
                else j++;
            }
        }

        return vector(ans.begin(),ans.end());
    }
};

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        int i=0;
        ListNode* mover = head;
        while(mover!=NULL){
            i++;
            mover=mover->next;
        }
        if(k>i) return head;
        int n = i;
        i=0;
        ListNode* second = NULL; // end of k node
        ListNode* first = head; // start of k node
        ListNode* prev = NULL;
        ListNode* curr = head;
        ListNode* next = NULL;
        while(curr!=NULL){
            i++;
            next = curr->next;
            curr->next=prev;
            prev=curr;
            curr=next;
            if(i%k==0){
                if(i==k) head = prev;
                first->next=curr;
                if(second!=NULL) second->next = prev;
                second=first;
                first=curr;
                prev=NULL;
                if(i==n-(n%k)) break;
            }
        }
        return head;
    }
};

    TreeNode* deleteNode(TreeNode* root, int key) {
        if(!root) return NULL;
        if(root->val==key){
            if(root->left && root->right){
                TreeNode* rightChild = root->right;
                TreeNode* lastRight = lastRightOfLeft(root->left);
                lastRight->right = rightChild;
                root = root->left;
            }else if(root->left){
                root = root->left;
            }else if(root->right){
                root = root->right;
            }else root = NULL;
        }else if(root->val<key){
            root->right = deleteNode(root->right,key);
        }
        else root->left = deleteNode(root->left,key);
        return root;
    }
