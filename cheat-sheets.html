🔀 Git Commands Version Control

git init                          # Initialize new repo
git clone <url>                   # Clone remote repo
git add .                         # Stage all changes
git add <file>                    # Stage specific file
git commit -m "message"           # Commit staged changes
git push origin main              # Push to remote
git pull origin main              # Pull from remote
git status                        # Check working tree
git log --oneline                 # Compact history
git diff                          # Show unstaged changes
git branch <name>                 # Create branch
git checkout -b <name>            # Create & switch branch
git merge <branch>                # Merge branch
git stash                         # Save uncommitted changes
git stash pop                     # Restore stashed changes
git reset --soft HEAD~1           # Undo last commit (keep changes)
git revert <hash>                 # Undo commit (safe for shared)

docker compose down               # Stop all services

docker cp <ctr>:/path ./host      # Copy file from container

ping host                         # Network test
curl url                          # HTTP request

🐘 SQL Commands Database

-- DDL
CREATE TABLE t (id INT PRIMARY KEY, name TEXT);
ALTER TABLE t ADD col TYPE;
DROP TABLE t;

-- DML
INSERT INTO t VALUES (1, 'name');
UPDATE t SET name='new' WHERE id=1;
DELETE FROM t WHERE id=1;

-- DQL
SELECT * FROM t WHERE id=1 ORDER BY name ASC LIMIT 10;
SELECT col1, COUNT(*) FROM t GROUP BY col1 HAVING COUNT(*)>1;

-- Joins
SELECT * FROM a INNER JOIN b ON a.id=b.aid;
SELECT * FROM a LEFT JOIN b ON a.id=b.aid;

-- Aggregate
SELECT COUNT(*), SUM(col), AVG(col), MIN(col), MAX(col) FROM t;

-- Index
CREATE INDEX idx ON t(col);

-- Transaction
BEGIN; UPDATE t SET col=val; COMMIT; -- or ROLLBACK;

🔧 Jenkins Pipeline CI/CD

pipeline {
  agent any
  stages {
    stage('Checkout') {
      steps { git 'https://github.com/user/repo.git' }
    }
    stage('Build') {
      steps { sh 'mvn clean package' }
    }
    stage('Test') {
      steps { sh 'mvn test' }
    }
    stage('Docker Build') {
      steps { sh 'docker build -t myapp .' }
    }
    stage('Deploy') {
      steps { sh 'docker push myapp:latest' }
    }
  }
  post {
    always { echo 'Pipeline finished' }
    success { echo 'Build succeeded!' }
    failure { echo 'Build failed!' }
  }
}
