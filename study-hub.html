🔄 Shell Scripting

#!/bin/bash
# Variables
export b=10          # Global variable
echo $b              # Print variable

# Read input
read -p "Enter value: " a
echo "$a is entered"

# Math operations
c=$((a+b))
echo "Sum: $c"

# If-Else
if [[ $a -gt 0 ]]; then
    echo "Positive"
else
    echo "Negative"
fi

# For loop
for (( i=1; i<=10; i++ )); do
    res=$(($a * $i))
    echo "$a * $i = $res"
done

# While loop
rem=0; res=0
while [[ $num -gt 0 ]]; do
    rem=$((num % 10))
    res=$((res * 10 + rem))
    num=$((num / 10))
done
echo "Reverse: $res"

# Case statement
case $num in
    1) echo "Sunday" ;;
    2) echo "Monday" ;;
    *) echo "Invalid" ;;
esac

🔢 KMP Pattern Matching — O(n+m)

string removeOccurrences(string s, string part) {
    int i = 0, j = 0, ind = -1;
    int n = part.length();
    while(i < s.length() && j < n) {
        if(s[i] == part[j]) {
            if(ind == -1) ind = i;
            if(j == n-1) { s.erase(ind, n); i=0; j=0; ind=-1; }
            else { i++; j++; }
        } else {
            if(j > 0) { i=ind+1; j=0; ind=-1; }
            else i++;
        }
    }
    return s;
}

🌲 BST Delete Node

TreeNode* lastRightOfLeft(TreeNode* root) {
    if(root->right == NULL) return root;
    return lastRightOfLeft(root->right);
}
TreeNode* deleteNode(TreeNode* root, int key) {
    if(!root) return NULL;
    if(root->val == key) {
        if(root->left && root->right) {
            TreeNode* rightChild = root->right;
            TreeNode* lastRight = lastRightOfLeft(root->left);
            lastRight->right = rightChild;
            root = root->left;
        } else if(root->left) root = root->left;
        else if(root->right) root = root->right;
        else root = NULL;
    } else if(root->val < key) root->right = deleteNode(root->right, key);
    else root->left = deleteNode(root->left, key);
    return root;
}

🌲 BST Kth Smallest

void in(TreeNode* root, int& i, int k, int& ans) {
    if(!root) return;
    in(root->left, i, k, ans);
    i++;
    if(i == k) ans = root->val;
    in(root->right, i, k, ans);
}
int kthSmallest(TreeNode* root, int k) {
    int ans = 0, i = 0;
    in(root, i, k, ans);
    return ans;
}

🔗 Reverse K-Group Linked List

ListNode* reverseKGroup(ListNode* head, int k) {
    int n = 0;
    ListNode* mover = head;
    while(mover) { n++; mover = mover->next; }
    if(k > n) return head;
    ListNode *second = NULL, *first = head, *prev = NULL, *curr = head, *next = NULL;
    int i = 0;
    while(curr) {
        i++; next = curr->next;
        curr->next = prev; prev = curr; curr = next;
        if(i % k == 0) {
            if(i == k) head = prev;
            first->next = curr;
            if(second) second->next = prev;
            second = first; first = curr; prev = NULL;
            if(i == n - (n % k)) break;
        }
    }
    return head;
}

📊 Binary Right Side View

void preorder(TreeNode* root, vector<int>& arr, int level) {
    if(root == NULL) return;
    if(arr.size() == level) arr.push_back(root->val);
    preorder(root->right, arr, level+1);
    preorder(root->left, arr, level+1);
}
vector<int> rightSideView(TreeNode* root) {
    vector<int> arr;
    preorder(root, arr, 0);
    return arr;
}
