Polymorphism

// Compile-time (Overloading)
int add(int a, int b);
double add(double a, double b);

// Runtime (Overriding)
class Parent {
    virtual void show() { }
};
class Child : public Parent {
    void show() override { }
};
Parent *ptr = new Child();
ptr->show();  // calls Child's show

Abstract Class

class Shape {
public:
    virtual void area() = 0; // Pure virtual
};
class Circle : public Shape {
    int r;
public:
    Circle(int r) : r(r) {}
    void area() override {
        cout << 3.14 * r * r;
    }
};

📊 Data Structures & Algorithms

Sorting Algorithms

// Bubble Sort - O(n²)
for(int i=0; iarr[j+1]) swap(arr[j],arr[j+1]);

// Selection Sort - O(n²)
for(int i=0; iarr[j]) min = j;
  swap(arr[i], arr[min]);
}

// Insertion Sort - O(n²)
for(int i=1; i=0 && arr[j]>key) {
    arr[j+1] = arr[j]; j--;
  }
  arr[j+1] = key;
}

Binary Search - O(log n)

int low=0, high=n-1;
while(low <= high) {
    int mid = (low+high)/2;
    if(arr[mid] == target) { found; }
    else if(arr[mid] < target) low = mid+1;
    else high = mid-1;
}

Linked List

class Node {
public:
    int data;
    Node *next;
    Node(int val) : data(val), next(NULL) {}
};

// Insert at head
void insertHead(Node *&head, int val) {
    Node *temp = new Node(val);
    temp->next = head;
    head = temp;
}

// Insert at tail
void insertTail(Node *&tail, int val) {
    Node *temp = new Node(val);
    tail->next = temp;
    tail = temp;
}

BST Traversal

class Node {
public:
    int data;
    Node *left, *right;
    Node(int val) : data(val), left(NULL), right(NULL) {}
};

// Pre-order: Root → Left → Right
void pre(Node *root) {
    if(!root) return;
    cout << root->data;
    pre(root->left); pre(root->right);
}

// In-order: Left → Root → Right
void in(Node *root) {
    if(!root) return;
    in(root->left);
    cout << root->data;
    in(root->right);
}

// Post-order: Left → Right → Root
void post(Node *root) {
    if(!root) return;
    post(root->left); post(root->right);
    cout << root->data;
}

Graph BFS

vector<int> adj[n]; // adjacency list
bool vis[n] = {false};

void bfs(int start) {
    queue<int> q;
    q.push(start);
    vis[start] = true;
    while(!q.empty()) {
        int node = q.front();
        q.pop();
        cout << node << " ";
        for(int v : adj[node]) {
            if(!vis[v]) {
                vis[v] = true;
                q.push(v);
            }
        }
    }
}

Inheritance

    String name = "Parent";
    void show() {
        System.out.println(name);
    }
}

class Child extends Parent {
    int age = 10;
    void display() {
        System.out.println(name + " " + age);
    }
}

Child c = new Child();
c.show();    // inherited method
c.display();

Method Overloading

class Calculator {
    int add(int a, int b) {
        return a + b;
    }
    double add(double a, double b) {
        return a + b;
    }
    int add(int a, int b, int c) {
        return a + b + c;
    }

Abstract Class

abstract class Shape {
    abstract void area();
    void display() {
        System.out.println("Shape");
    }
}

class Circle extends Shape {
    int r;
    Circle(int r) { this.r = r; }
    void area() {
        System.out.println(3.14*r*r);
    }
}
